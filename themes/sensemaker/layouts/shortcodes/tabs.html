{{- $group := default (printf "tabs-%d" .Ordinal) (.Get "id") -}} {{- $raw := trim .Inner " \r\n\t" -}} {{- if eq $raw "" -}} {{- warnf "tabs shortcode on %q is empty" .Page.Path -}} {{- else -}} {{- $normalized := replaceRE "(?m)\r?\n---\r?\n" "\n[[TABS-SPLIT]]\n" $raw -}} {{- $sections := split $normalized "[[TABS-SPLIT]]" -}}
{{- range $index, $section := $sections -}} {{- $section = trim $section " \r\n\t" -}} {{- if ne $section "" -}} {{- $lines := split $section "\n" -}} {{- $title := trim (index $lines 0) " \r\n\t" -}} {{- $body := trim (delimit (after 1 $lines) "\n") " \r\n\t" -}} {{- $tabId := printf "%s-%d" $group $index -}}
{{ $body | markdownify }}
{{- end -}} {{- end -}}
{{- end -}}
