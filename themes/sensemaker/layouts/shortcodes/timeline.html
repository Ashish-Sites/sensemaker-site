{{- $group := default (printf "timeline-%d" .Ordinal) (.Get "id") -}} {{- $raw := trim .Inner " \r\n\t" -}} {{- if eq $raw "" -}} {{- warnf "timeline shortcode on %q is empty" .Page.Path -}} {{- else -}} {{- $normalized := replaceRE "(?m)\r?\n---\r?\n" "\n[[TIMELINE-SPLIT]]\n" $raw -}} {{- $sections := split $normalized "[[TIMELINE-SPLIT]]" -}}
    {{- range $section := $sections -}} {{- $section = trim $section " \r\n\t" -}} {{- if ne $section "" -}} {{- $lines := split $section "\n" -}} {{- $metaLine := trim (index $lines 0) " \r\n\t" -}} {{- $parts := split $metaLine "|" -}} {{- $date := trim (index $parts 0) " \r\n\t" -}} {{- $title := default "" (trim (index $parts 1) " \r\n\t") -}} {{- $body := trim (delimit (after 1 $lines) "\n") " \r\n\t" -}}
  1. {{- if ne $date "" }}{{ $date }}{{ end -}}
    {{- if ne $title "" }}

    {{ $title }}

    {{ end -}} {{- if ne $body "" }}
    {{ $body | markdownify }}
    {{ end -}}
    2. {{- end -}} {{- end -}}
{{- end -}}
